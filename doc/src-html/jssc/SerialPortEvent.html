025package jssc;
026
027/**
028 *
029 * @author scream3r
030 */
031public class SerialPortEvent {
032
033    private String portName;
034    private int eventType;
035    private int eventValue;
036
037    public static final int RXCHAR = 1;
038    public static final int RXFLAG = 2;
039    public static final int TXEMPTY = 4;
040    public static final int CTS = 8;
041    public static final int DSR = 16;
042    public static final int RLSD = 32;
043    public static final int BREAK = 64;
044    public static final int ERR = 128;
045    public static final int RING = 256;
046
047    public SerialPortEvent(String portName, int eventType, int eventValue){
048        this.portName = portName;
049        this.eventType = eventType;
050        this.eventValue = eventValue;
051    }
052
053    /**
054     * Getting port name which sent the event
055     */
056    public String getPortName() {
057        return portName;
058    }
059
060    /**
061     * Getting event type
062     */
063    public int getEventType() {
064        return eventType;
065    }
066
067    /**
068     * Getting event value
069     * <br></br>
070     * <br><u><b>Event values depending on their types:</b></u></br>
071     * <br><b>RXCHAR</b> - bytes count in input buffer</br>
072     * <br><b>RXFLAG</b> - bytes count in input buffer (Not supported in Linux)</br>
073     * <br><b>TXEMPTY</b> - bytes count in output buffer</br>
074     * <br><b>CTS</b> - state of CTS line (0 - OFF, 1 - ON)</br>
075     * <br><b>DSR</b> - state of DSR line (0 - OFF, 1 - ON)</br>
076     * <br><b>RLSD</b> - state of RLSD line (0 - OFF, 1 - ON)</br>
077     * <br><b>BREAK</b> - 0</br>
078     * <br><b>RING</b> - state of RING line (0 - OFF, 1 - ON)</br>
079     * <br><b>ERR</b> - mask of errors</br>
080     */
081    public int getEventValue() {
082        return eventValue;
083    }
084
085    /**
086     * Method returns true if event of type <b>"RXCHAR"</b> is received and otherwise false
087     */
088    public boolean isRXCHAR() {
089        if(eventType == RXCHAR){
090            return true;
091        }
092        else {
093            return false;
094        }
095    }
096
097    /**
098     * Method returns true if event of type <b>"RXFLAG"</b> is received and otherwise false
099     */
100    public boolean isRXFLAG() {
101        if(eventType == RXFLAG){
102            return true;
103        }
104        else {
105            return false;
106        }
107    }
108
109    /**
110     * Method returns true if event of type <b>"TXEMPTY"</b> is received and otherwise false
111     */
112    public boolean isTXEMPTY() {
113        if(eventType == TXEMPTY){
114            return true;
115        }
116        else {
117            return false;
118        }
119    }
120
121    /**
122     * Method returns true if event of type <b>"CTS"</b> is received and otherwise false
123     */
124    public boolean isCTS() {
125        if(eventType == CTS){
126            return true;
127        }
128        else {
129            return false;
130        }
131    }
132
133    /**
134     * Method returns true if event of type <b>"DSR"</b> is received and otherwise false
135     */
136    public boolean isDSR() {
137        if(eventType == DSR){
138            return true;
139        }
140        else {
141            return false;
142        }
143    }
144
145    /**
146     * Method returns true if event of type <b>"RLSD"</b> is received and otherwise false
147     */
148    public boolean isRLSD() {
149        if(eventType == RLSD){
150            return true;
151        }
152        else {
153            return false;
154        }
155    }
156
157    /**
158     * Method returns true if event of type <b>"BREAK"</b> is received and otherwise false
159     */
160    public boolean isBREAK() {
161        if(eventType == BREAK){
162            return true;
163        }
164        else {
165            return false;
166        }
167    }
168
169    /**
170     * Method returns true if event of type <b>"ERR"</b> is received and otherwise false
171     */
172    public boolean isERR() {
173        if(eventType == ERR){
174            return true;
175        }
176        else {
177            return false;
178        }
179    }
180
181    /**
182     * Method returns true if event of type <b>"RING"</b> is received and otherwise false
183     */
184    public boolean isRING() {
185        if(eventType == RING){
186            return true;
187        }
188        else {
189            return false;
190        }
191    }
192}
