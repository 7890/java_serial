025package jssc;
026
027/**
028 *
029 * @author scream3r
030 */
031public class SerialPortException extends Exception {
032
033    final public static String TYPE_PORT_ALREADY_OPENED = "Port already opened";
034    final public static String TYPE_PORT_NOT_OPENED = "Port not opened";
035    final public static String TYPE_CANT_SET_MASK = "Can't set mask";
036    final public static String TYPE_LISTENER_ALREADY_ADDED = "Event listener already added";
037    final public static String TYPE_LISTENER_THREAD_INTERRUPTED = "Event listener thread interrupted";
038    final public static String TYPE_CANT_REMOVE_LISTENER = "Can't remove event listener, because listener not added";
039    /**
040     * @since 0.8
041     */
042    final public static String TYPE_PARAMETER_IS_NOT_CORRECT = "Parameter is not correct";
043    /**
044     * @since 0.8
045     */
046    final public static String TYPE_NULL_NOT_PERMITTED = "Null not permitted";
047    /**
048     * @since 0.9.0
049     */
050    final public static String TYPE_PORT_BUSY = "Port busy";
051    /**
052     * @since 0.9.0
053     */
054    final public static String TYPE_PORT_NOT_FOUND = "Port not found";
055    /**
056     * @since 2.2.0
057     */
058    final public static String TYPE_PERMISSION_DENIED = "Permission denied";
059    /**
060     * @since 2.3.0
061     */
062    final public static String TYPE_INCORRECT_SERIAL_PORT = "Incorrect serial port";
063
064    private String portName;
065    private String methodName;
066    private String exceptionType;
067
068    public SerialPortException(String portName, String methodName, String exceptionType){
069        super("Port name - " + portName + "; Method name - " + methodName + "; Exception type - " + exceptionType + ".");
070        this.portName = portName;
071        this.methodName = methodName;
072        this.exceptionType = exceptionType;
073    }
074
075    /**
076     * Getting port name during operation with which the exception was called
077     */
078    public String getPortName(){
079        return portName;
080    }
081
082    /**
083     * Getting method name during execution of which the exception was called
084     */
085    public String getMethodName(){
086        return methodName;
087    }
088
089    /**
090     * Getting exception type
091     */
092    public String getExceptionType(){
093        return exceptionType;
094    }
095}
