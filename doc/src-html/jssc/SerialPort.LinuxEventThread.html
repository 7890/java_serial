141    /**
142     * Port opening
143     * <br><br>
144     * <b>Note: </b>If port busy <b>TYPE_PORT_BUSY</b> exception will be thrown.
145     * If port not found <b>TYPE_PORT_NOT_FOUND</b> exception will be thrown.
146     *
147     * @return If the operation is successfully completed, the method returns true
148     *
149     * @throws SerialPortException
150     */
151    public boolean openPort() throws SerialPortException {
152        if(portOpened){
153            throw new SerialPortException(portName, "openPort()", SerialPortException.TYPE_PORT_ALREADY_OPENED);
154        }
155        if(portName != null){
156            boolean useTIOCEXCL = (System.getProperty(SerialNativeInterface.PROPERTY_JSSC_NO_TIOCEXCL) == null &&
157                                   System.getProperty(SerialNativeInterface.PROPERTY_JSSC_NO_TIOCEXCL.toLowerCase()) == null);
158            portHandle = serialInterface.openPort(portName, useTIOCEXCL);//since 2.3.0 -> (if JSSC_NO_TIOCEXCL defined, exclusive lock for serial port will be disabled)
159        }
160        else {
161            throw new SerialPortException(portName, "openPort()", SerialPortException.TYPE_NULL_NOT_PERMITTED);//since 2.1.0 -> NULL port name fix
162        }
163        if(portHandle == SerialNativeInterface.ERR_PORT_BUSY){
164            throw new SerialPortException(portName, "openPort()", SerialPortException.TYPE_PORT_BUSY);
165        }
166        else if(portHandle == SerialNativeInterface.ERR_PORT_NOT_FOUND){
167            throw new SerialPortException(portName, "openPort()", SerialPortException.TYPE_PORT_NOT_FOUND);
168        }
169        else if(portHandle == SerialNativeInterface.ERR_PERMISSION_DENIED){
170            throw new SerialPortException(portName, "openPort()", SerialPortException.TYPE_PERMISSION_DENIED);
171        }
172        else if(portHandle == SerialNativeInterface.ERR_INCORRECT_SERIAL_PORT){
173            throw new SerialPortException(portName, "openPort()", SerialPortException.TYPE_INCORRECT_SERIAL_PORT);
174        }
175        portOpened = true;
176        return true;
177    }

195    /**
196     * Setting the parameters of port
197     *
198     * @param baudRate data transfer rate
199     * @param dataBits number of data bits
200     * @param stopBits number of stop bits
201     * @param parity parity
202     * @param setRTS initial state of RTS line(ON/OFF)
203     * @param setDTR initial state of DTR line(ON/OFF)
204     *
205     * @return If the operation is successfully completed, the method returns true, otherwise false
206     *
207     * @throws SerialPortException
208     *
209     * @since 0.8
210     */
211    public boolean setParams(int baudRate, int dataBits, int stopBits, int parity, boolean setRTS, boolean setDTR) throws SerialPortException {
212        checkPortOpened("setParams()");
213        if(stopBits == 1){
214            stopBits = 0;
215        }
216        else if(stopBits == 3){
217            stopBits = 1;
218        }
219        int flags = 0;
220        if(System.getProperty(SerialNativeInterface.PROPERTY_JSSC_IGNPAR) != null || System.getProperty(SerialNativeInterface.PROPERTY_JSSC_IGNPAR.toLowerCase()) != null){
221            flags |= PARAMS_FLAG_IGNPAR;
222        }
223        if(System.getProperty(SerialNativeInterface.PROPERTY_JSSC_PARMRK) != null || System.getProperty(SerialNativeInterface.PROPERTY_JSSC_PARMRK.toLowerCase()) != null){
224            flags |= PARAMS_FLAG_PARMRK;
225        }
226        return serialInterface.setParams(portHandle, baudRate, dataBits, stopBits, parity, setRTS, setDTR, flags);
227    }

252    /**
253     * Set events mask. Required flags shall be sent to the input. Variables with prefix 
254     * <b>"MASK_"</b>, shall be used as flags, for example <b>"MASK_RXCHAR"</b>. 
255     * Sent parameter "mask" is additive value, so addition of flags is allowed.
256     * For example if messages about data receipt and CTS and DSR status changing
257     * shall be received, it is required to set the mask - <b>"MASK_RXCHAR | MASK_CTS | MASK_DSR"</b>
258     * 
259     * @return If the operation is successfully completed, the method returns true, otherwise false
260     *
261     * @throws SerialPortException
262     */
263    public boolean setEventsMask(int mask) throws SerialPortException {
264        checkPortOpened("setEventsMask()");
265        if(SerialNativeInterface.getOsType() == SerialNativeInterface.OS_LINUX ||
266           SerialNativeInterface.getOsType() == SerialNativeInterface.OS_SOLARIS ||
267           SerialNativeInterface.getOsType() == SerialNativeInterface.OS_MAC_OS_X){//since 0.9.0
268            linuxMask = mask;
269            if(mask > 0){
270                maskAssigned = true;
271            }
272            else {
273                maskAssigned = false;
274            }
275            return true;
276        }
277        boolean returnValue = serialInterface.setEventsMask(portHandle, mask);
278        if(!returnValue){
279            throw new SerialPortException(portName, "setEventsMask()", SerialPortException.TYPE_CANT_SET_MASK);
280        }
281        if(mask > 0){
282            maskAssigned = true;
283        }
284        else {
285            maskAssigned = false;
286        }
287        return returnValue;
288    }

483    public String readHexString(int byteCount, String separator) throws SerialPortException {
484        checkPortOpened("readHexString()");
485        String[] strBuffer = readHexStringArray(byteCount);
486        String returnString = "";
487        boolean insertSeparator = false;
488        for(String value : strBuffer){
489            if(insertSeparator){
490                returnString += separator;
491            }
492            returnString += value;
493            insertSeparator = true;
494        }
495        return returnString;

501     * @param byteCount count of bytes for reading

513        for(int i = 0; i < intBuffer.length; i++){

518            strBuffer[i] = value;

549    private void waitBytesWithTimeout(String methodName, int byteCount, int timeout) throws SerialPortException, SerialPortTimeoutException {
550        checkPortOpened("waitBytesWithTimeout()");
551        boolean timeIsOut = true;
552        long startTime = System.currentTimeMillis();
553        while((System.currentTimeMillis() - startTime) < timeout){
554            if(getInputBufferBytesCount() >= byteCount){
555                timeIsOut = false;
556                break;
557            }
558            try {
559                Thread.sleep(0, 100);//Need to sleep some time to prevent high CPU loading
560            }
561            catch (InterruptedException ex) {
562                //Do nothing
563            }
564        }
565        if(timeIsOut){
566            throw new SerialPortTimeoutException(portName, methodName, timeout);
567        }
568    }

702    /**
703     * Read all available bytes from port like a String
704     *
705     * @return If input buffer is empty <b>null</b> will be returned, else byte array with all data from port converted to String

707     * @throws SerialPortException

712        checkPortOpened("readString()");

992    /**
993     * Internal method. Add event listener. Object of <b>"SerialPortEventListener"</b> type shall be sent
994     * to the method. This object shall be properly described, as it will be in
995     * charge for handling of occurred events. Also events mask shall be sent to
996     * this method, to do it use variables with prefix <b>"MASK_"</b> for example <b>"MASK_RXCHAR"</b>. If
997     * <b>overwriteMask == true</b> and mask has been already assigned it value will be rewrited by <b>mask</b>
998     * value, if <b>overwriteMask == false</b> and mask has been already assigned the new <b>mask</b> value will be ignored,
999     * if there is no assigned mask to this serial port the <b>mask</b> value will be used for setting it up in spite of
1000     * <b>overwriteMask</b> value
1001     *
1002     * @see #setEventsMask(int) setEventsMask(int mask)
1003     *
1004     * @throws SerialPortException
1005     */
1006    private void addEventListener(SerialPortEventListener listener, int mask, boolean overwriteMask) throws SerialPortException {
1007        checkPortOpened("addEventListener()");
1008        if(!eventListenerAdded){
1009            if((maskAssigned && overwriteMask) || !maskAssigned) {
1010                setEventsMask(mask);
1011            }
1012            eventListener = listener;
1013            eventThread = getNewEventThread();
1014            eventThread.setName("EventThread " + portName);
1015            //since 2.2.0 ->
1016            try {
1017                Method method = eventListener.getClass().getMethod("errorOccurred", new Class[]{SerialPortException.class});
1018                method.setAccessible(true);
1019                methodErrorOccurred = method;
1020            }
1021            catch (SecurityException ex) {
1022                //Do nothing
1023            }
1024            catch (NoSuchMethodException ex) {
1025                //Do nothing
1026            }
1027            //<- since 2.2.0
1028            eventThread.start();
1029            eventListenerAdded = true;
1030        }
1031        else {
1032            throw new SerialPortException(portName, "addEventListener()", SerialPortException.TYPE_LISTENER_ALREADY_ADDED);
1033        }
1034    }

1050    /**
1051     * Delete event listener. Mask is set to 0. So at the next addition of event
1052     * handler you shall set required event mask again
1053     *
1054     * @return If the operation is successfully completed, the method returns true, otherwise false
1055     * 
1056     * @throws SerialPortException
1057     */
1058    public boolean removeEventListener() throws SerialPortException {
1059        checkPortOpened("removeEventListener()");
1060        if(!eventListenerAdded){
1061            throw new SerialPortException(portName, "removeEventListener()", SerialPortException.TYPE_CANT_REMOVE_LISTENER);
1062        }
1063        eventThread.terminateThread();
1064        setEventsMask(0);
1065        if(Thread.currentThread().getId() != eventThread.getId()){
1066            if(eventThread.isAlive()){
1067                try {
1068                    eventThread.join(5000);
1069                }
1070                catch (InterruptedException ex) {
1071                    throw new SerialPortException(portName, "removeEventListener()", SerialPortException.TYPE_LISTENER_THREAD_INTERRUPTED);
1072                }
1073            }
1074        }
1075        methodErrorOccurred = null;
1076        eventListenerAdded = false;
1077        return true;
1078    }

1132    /**
1133     * EventListener for Linux OS
1134     *
1135     * @since 0.8
1136     */
1137    private class LinuxEventThread extends EventThread {
1138
1139        //Essential interruptions for events: BREAK, ERR, TXEMPTY
1140        private final int INTERRUPT_BREAK = 512;
1141        private final int INTERRUPT_TX = 1024;
1142        private final int INTERRUPT_FRAME = 2048;
1143        private final int INTERRUPT_OVERRUN = 4096;
1144        private final int INTERRUPT_PARITY = 8192;
1145
1146        //Count of interruptions
1147        private int interruptBreak;
1148        private int interruptTX;
1149        private int interruptFrame;
1150        private int interruptOverrun;
1151        private int interruptParity;
1152
1153        //Previous states if lines (then state change event will be generated)
1154        private int preCTS;
1155        private int preDSR;
1156        private int preRLSD;
1157        private int preRING;
1158
1159        //Need to get initial states
1160        public LinuxEventThread(){
1161            int[][] eventArray = waitEvents();
1162            for(int i = 0; i < eventArray.length; i++){
1163                int eventType = eventArray[i][0];
1164                int eventValue = eventArray[i][1];
1165                switch(eventType){
1166                    case INTERRUPT_BREAK:
1167                        interruptBreak = eventValue;
1168                        break;
1169                    case INTERRUPT_TX:
1170                        interruptTX = eventValue;
1171                        break;
1172                    case INTERRUPT_FRAME:
1173                        interruptFrame = eventValue;
1174                        break;
1175                    case INTERRUPT_OVERRUN:
1176                        interruptOverrun = eventValue;
1177                        break;
1178                    case INTERRUPT_PARITY:
1179                        interruptParity = eventValue;
1180                        break;
1181                    case MASK_CTS:
1182                        preCTS = eventValue;
1183                        break;
1184                    case MASK_DSR:
1185                        preDSR = eventValue;
1186                        break;
1187                    case MASK_RING:
1188                        preRING = eventValue;
1189                        break;
1190                    case MASK_RLSD:
1191                        preRLSD = eventValue;
1192                        break;
1193                }
1194            }
1195        }
1196
1197        @Override
1198        public void run() {
1199            while(!super.threadTerminated){
1200                int[][] eventArray = waitEvents();
1201                int mask = getLinuxMask();
1202                boolean interruptTxChanged = false;
1203                int errorMask = 0;
1204                for(int i = 0; i < eventArray.length; i++){
1205                    boolean sendEvent = false;
1206                    int eventType = eventArray[i][0];
1207                    int eventValue = eventArray[i][1];
1208                    if(eventType > 0 && !super.threadTerminated){
1209                        switch(eventType){
1210                            case INTERRUPT_BREAK:
1211                                if(eventValue != interruptBreak){
1212                                    interruptBreak = eventValue;
1213                                    if((mask & MASK_BREAK) == MASK_BREAK){
1214                                        eventType = MASK_BREAK;
1215                                        eventValue = 0;
1216                                        sendEvent = true;
1217                                    }
1218                                }
1219                                break;
1220                            case INTERRUPT_TX:
1221                                if(eventValue != interruptTX){
1222                                    interruptTX = eventValue;
1223                                    interruptTxChanged = true;
1224                                }
1225                                break;
1226                            case INTERRUPT_FRAME:
1227                                if(eventValue != interruptFrame){
1228                                    interruptFrame = eventValue;
1229                                    errorMask |= ERROR_FRAME;
1230                                }
1231                                break;
1232                            case INTERRUPT_OVERRUN:
1233                                if(eventValue != interruptOverrun){
1234                                    interruptOverrun = eventValue;
1235                                    errorMask |= ERROR_OVERRUN;
1236                                }
1237                                break;
1238                            case INTERRUPT_PARITY:
1239                                if(eventValue != interruptParity){
1240                                    interruptParity = eventValue;
1241                                    errorMask |= ERROR_PARITY;
1242                                }
1243                                if((mask & MASK_ERR) == MASK_ERR && errorMask != 0){
1244                                    eventType = MASK_ERR;
1245                                    eventValue = errorMask;
1246                                    sendEvent = true;
1247                                }
1248                                break;
1249                            case MASK_CTS:
1250                                if(eventValue != preCTS){
1251                                    preCTS = eventValue;
1252                                    if((mask & MASK_CTS) == MASK_CTS){
1253                                        sendEvent = true;
1254                                    }
1255                                }
1256                                break;
1257                            case MASK_DSR:
1258                                if(eventValue != preDSR){
1259                                    preDSR = eventValue;
1260                                    if((mask & MASK_DSR) == MASK_DSR){
1261                                        sendEvent = true;
1262                                    }
1263                                }
1264                                break;
1265                            case MASK_RING:
1266                                if(eventValue != preRING){
1267                                    preRING = eventValue;
1268                                    if((mask & MASK_RING) == MASK_RING){
1269                                        sendEvent = true;
1270                                    }
1271                                }
1272                                break;
1273                            case MASK_RLSD: /*DCD*/
1274                                if(eventValue != preRLSD){
1275                                    preRLSD = eventValue;
1276                                    if((mask & MASK_RLSD) == MASK_RLSD){
1277                                        sendEvent = true;
1278                                    }
1279                                }
1280                                break;
1281                            case MASK_RXCHAR:
1282                                if(((mask & MASK_RXCHAR) == MASK_RXCHAR) && (eventValue > 0)){
1283                                    sendEvent = true;
1284                                }
1285                                break;
1286                            /*case MASK_RXFLAG:
1287                                //Do nothing at this moment
1288                                if(((mask & MASK_RXFLAG) == MASK_RXFLAG) && (eventValue > 0)){
1289                                    sendEvent = true;
1290                                }
1291                                break;*/
1292                            case MASK_TXEMPTY:
1293                                if(((mask & MASK_TXEMPTY) == MASK_TXEMPTY) && (eventValue == 0) && interruptTxChanged){
1294                                    sendEvent = true;
1295                                }
1296                                break;
1297                        }
1298                        if(sendEvent){
1299                            eventListener.serialEvent(new SerialPortEvent(portName, eventType, eventValue));
1300                        }
1301                    }
1302                }
1303                //Need to sleep some time
1304                try {
1305                    Thread.sleep(0, 100);
1306                }
1307                catch (Exception ex) {
1308                    //Do nothing
1309                }
1310            }
1311        }
